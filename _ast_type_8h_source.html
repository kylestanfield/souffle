56 /**
57  * A primitive type is named type that can either be a sub-type of
58  * the build-in number or symbol type. Primitive types are the most
59  * basic building blocks of souffle's type system.
60  */

64  AstPrimitiveType(const AstQualifiedName& name, TypeAttribute type) : AstType(name), type(type) {}

145 /**
146  * A record type aggregates a list of fields into a new type.
147  * Each record type has a name making it unique. Two record
148  * types are unrelated to all other types (they do not have
149  * any super or sub types).
150  */
151 class AstRecordType : public AstType {
152 public:

154  struct Field {
155  std::string name; // < the field name
156  AstQualifiedName type; // < the field type
157

159  return this == &other || (name == other.name && type == other.type);
160  }
161  };

165  fields.push_back(Field({name, type}));

170  return fields;

175  assert(idx < fields.size() && "record field index out of bounds");
176  fields[idx].type = type;

179  AstRecordType* clone() const override {
180  auto res = new AstRecordType();

182  res->setQualifiedName(getQualifiedName());
183  res->fields = fields;
184  return res;

186 
187 protected:
